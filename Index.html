<script>

  const DRINKS = {

    bier: { emoji: '🍺', name: 'Bier',        price: 18, short: 'Bier' },

    wein: { emoji: '🍷', name: 'Wein',        price: 18, short: 'Wein' },

    af:   { emoji: '🥤', name: 'Alkoholfrei', price: 12, short: 'AF'   }

  };


  let personDrinks = [];


  function onAttendingChange() {

    const val = document.querySelector('input[name="attending"]:checked')?.value;

    document.getElementById('personsCard').style.display = val === 'ja' ? 'block' : 'none';

    document.getElementById('nameCard').style.display = 'none';

    document.getElementById('submitBtn').style.display = 'none';

  }


  function onMehr() {

    const inp = document.getElementById('moreInput');

    inp.style.display = 'block';

    inp.focus();

    document.getElementById('drinkSection').classList.remove('visible');

    document.getElementById('drinkRows').innerHTML = '';

    personDrinks = [];

  }


  function buildDrinks(count) {

    const section = document.getElementById('drinkSection');

    const rows = document.getElementById('drinkRows');


    if (!count || count < 1) {

      section.classList.remove('visible');

      personDrinks = [];

      return;

    }


    // preserve existing picks

    while (personDrinks.length < count) personDrinks.push(null);

    personDrinks = personDrinks.slice(0, count);


    rows.innerHTML = '';

    for (let i = 0; i < count; i++) {

      const row = document.createElement('div');

      row.className = 'person-drink-row';

      row.innerHTML = `

        <div class="person-label">Person ${i + 1}${i === 0 ? ' (du)' : ''}</div>

        <div class="drink-btns">

          ${Object.entries(DRINKS).map(([key, d]) => `

            <button type="button"

              class="drink-btn${personDrinks[i] === key ? ' selected' : ''}"

              onclick="pickDrink(${i},'${key}',this)">

              ${d.emoji} ${d.name} <span class="ptag">${d.price}€</span>

            </button>

          `).join('')}

        </div>

      `;

      rows.appendChild(row);

    }


    section.classList.add('visible');

    checkFormReady();

  }


  function pickDrink(idx, key, btn) {

    personDrinks[idx] = key;

    const row = btn.closest('.person-drink-row');

    row.querySelectorAll('.drink-btn').forEach(b => b.classList.remove('selected'));

    btn.classList.add('selected');

    checkFormReady();

  }


  function getCount() {

    const sel = document.querySelector('input[name="persons"]:checked');

    if (!sel) return 0;

    if (sel.value === 'mehr') return parseInt(document.getElementById('moreInput').value) || 0;

    return parseInt(sel.value);

  }


  function checkFormReady() {

    const count = getCount();

    if (count < 1) return;

    const allPicked = personDrinks.slice(0, count).every(d => d !== null);

    document.getElementById('nameCard').style.display = allPicked ? 'block' : 'none';

    document.getElementById('submitBtn').style.display = allPicked ? 'block' : 'none';

  }


  function handleSubmit(e) {

    e.preventDefault();


    const attending = document.querySelector('input[name="attending"]:checked')?.value;

    if (attending !== 'ja') {

      alert('Du hast "Nein" ausgewählt – keine Zahlung nötig. Bis zum nächsten Mal! 👋');

      return;

    }


    const count = getCount();

    if (count < 1) { alert('Bitte Personenanzahl auswählen.'); return; }


    const missing = personDrinks.slice(0, count).findIndex(d => !d);

    if (missing !== -1) { alert(`Bitte für Person ${missing + 1} ein Getränk auswählen.`); return; }


    const name = document.getElementById('nameInput').value.trim();

    if (!name) { alert('Bitte deinen Namen eintragen.'); return; }


    // Calculate total

    const drinks = personDrinks.slice(0, count);

    const total = drinks.reduce((s, k) => s + DRINKS[k].price, 0);


    // Build drink counts for subject

    const counts = {};

    drinks.forEach(k => counts[k] = (counts[k] || 0) + 1);

    const drinkStr = Object.entries(counts).map(([k, n]) => `${n}x${DRINKS[k].short}`).join('+');


    // PayPal subject: name | 2P | 1xBier+1xAF | 30EUR

    const subject = `${name} | ${count}P | ${drinkStr} | ${total}EUR`;

    const paypalUrl = `https://www.paypal.me/Snuscheck/${total}EUR?locale.x=de_DE`;

    // Note: paypal.me doesn't support custom note in URL natively, but we show subject clearly


    // Render summary

    const rows = document.getElementById('summaryRows');

    rows.innerHTML = `

      <div class="summary-row"><span class="s-label">👤 Name</span><span class="s-value">${name}</span></div>

      <div class="summary-row"><span class="s-label">👥 Personen</span><span class="s-value">${count}</span></div>

      ${drinks.map((k, i) => `

        <div class="summary-row">
